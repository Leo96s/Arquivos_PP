package projeto_pp;

/**
 * The Edition interface represents an edition in Challenge Based Learning (CBL).
 * An edition is a specific instance of CBL with a unique ID and duration.
 * @author Leonardo
 */
public interface Edition {
    
    
    /**
     * Returns the ID of the edition.
     *
     * @return the ID of the edition
     */
    public int getId();
    
    /**
     * Returns the title of the edition.
     *
     * @return the title of the edition
     */
    public String getTitle();
    
    /**
     * Returns the description of the edition.
     *
     * @return the description of the edition
     */
    public String getDescription();
    
    /**
     * Returns the start date of the edition.
     *
     * @return the start date of the edition
     */
    public String getStartDate();
    
    /**
     * Returns the end date of the edition.
     *
     * @return the end date of the edition
     */
    public String getEndDate();
}
