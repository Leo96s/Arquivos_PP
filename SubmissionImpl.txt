package projeto_pp;

/**
 * Represents a submission made by a student for a task in Challenge Based Learning (CBL).
 */
public class SubmissionImpl {
    private Task task;
    private String studentName;
    private String submissionDate;
    private String content;

    /**
     * Constructs a new Submission object with the specified task, student name, and content.
     *
     * @param task         the task associated with the submission
     * @param studentName  the name of the student making the submission
     * @param content      the content of the submission
     */
    public SubmissionImpl(Task task, String studentName, String submissionDate, String content) {
        this.task = task;
        this.studentName = studentName;
        this.submissionDate = submissionDate;
        this.content = content;
    }

    /**
     * Returns the task associated with the submission.
     *
     * @return the task associated with the submission
     */
    public Task getTask() {
        return task;
    }

    /**
     * Sets the task associated with the submission.
     *
     * @param task the task to be associated with the submission
     */
    public void setTask(Task task) {
        this.task = task;
    }

    /**
     * Returns the name of the student who made the submission.
     *
     * @return the name of the student who made the submission
     */
    public String getStudentName() {
        return studentName;
    }

    /**
     * Sets the name of the student who made the submission.
     *
     * @param studentName the name of the student who made the submission
     */
    public void setStudentName(String studentName) {
        this.studentName = studentName;
    }

    /**
     * Returns the date and time when the submission was made.
     *
     * @return the date and time when the submission was made
     */
    public String getSubmissionDate() {
        return submissionDate;
    }

    /**
     * Sets the date and time when the submission was made.
     *
     * @param submissionDate the date and time when the submission was made
     */
    public void setSubmissionDate(String submissionDate) {
        this.submissionDate = submissionDate;
    }

    /**
     * Returns the content of the submission.
     *
     * @return the content of the submission
     */
    public String getContent() {
        return content;
    }

    /**
     * Sets the content of the submission.
     *
     * @param content the content of the submission
     */
    public void setContent(String content) {
        this.content = content;
    }

    /**
     * Returns a string representation of the Submission object.
     *
     * @return a string representation of the Submission object
     */
    @Override
    public String toString() {
        return "Submission{" +
                "task=" + task +
                ", studentName='" + studentName + '\'' +
                ", submissionDate=" + submissionDate +
                ", content='" + content + '\'' +
                '}';
    }
}
